### 1. No Merge

```terraform
# main.tf

output "c_simplemapvariable_7" {
  value = [
    for i in local.test2: {
        for sn in i.subnets: format("%s_%s", i.peering_connection_id, sn) => {
          connection_id = i.peering_connection_id
          subnet = sn
        }
        
    }
  ]
}
```

### 2. With Merge

```terraform
# main.tf
output "c_simplemapvariable_7" {
  value = merge([
    for i in local.test2: {
        for sn in i.subnets: format("%s_%s", i.peering_connection_id, sn) => {
          connection_id = i.peering_connection_id
          subnet = sn
        }
        
    }
  ]...)
}

```
